% Image Dimensions
% 1. images are set with a default width and height. Because 'keepaspectratio' is set,
% an image will be scaled unitl it hits the first lowest threshold.
% 2. Image dimensions can be overwritten locally for each image. If only width is changed,
% the height will be automatically adjusted to maintain the aspect ratio.
%* Therefore it is recommended to only specfiy the width of an image, and let the height adjust automatically. *

% image width can be adjusted. Together with 'keepaspectratio' the height will be auto adjusted.
\newlength{\imagewidth}
\newlength{\maximagewidth}
\setlength{\imagewidth}{0.5\columnwidth}
\setlength{\maximagewidth}{1.0\columnwidth}

% upper bound for image heigths.
\newlength{\maximageheight}
\setlength{\maximageheight}{0.4\textheight}
